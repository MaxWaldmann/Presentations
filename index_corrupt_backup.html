   function Dijkstra(Graph, source):
    
       for each vertex v in Graph.Vertices:
           dist[v] ← INFINITY
           prev[v] ← UNDEFINED
           add v to Q
      dist[source] ← 0
     
       while Q is not empty:
          u ← vertex in Q with minimum dist[u]
          Q.remove(u)
         
          for each edge (u, v) in Graph:
              alt ← dist[u] + Graph.Distance(u,v)
              if alt < dist[v]:
                  dist[v] ← alt
                  prev[v] ← u

      return dist[], prev[]
